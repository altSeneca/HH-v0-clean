Implementation Sample

// Camera Screen with Project Selector
@Composable
fun SmartCameraScreen(
    viewModel: CameraViewModel = hiltViewModel()
) {
    val context = LocalContext.current
    val cameraController = remember { LifecycleCameraController(context) }
    val photoState by viewModel.photoState.collectAsState()
    
    Column(modifier = Modifier.fillMaxSize()) {
        // Project Selector at Top
        ProjectQuickSelector(
            currentProject = photoState.currentProject,
            onProjectChange = { viewModel.setProject(it) }
        )
        
        Box(modifier = Modifier.fillMaxSize()) {
            // Camera Preview
            CameraPreview(
                controller = cameraController,
                modifier = Modifier.fillMaxSize()
            )
            
            // Metadata Overlay (includes selected project)
            MetadataOverlay(
                location = photoState.currentLocation,
                project = photoState.currentProject,
                timestamp = photoState.timestamp
            )
            
            // Capture Button
            CaptureButton(
                onClick = {
                    capturePhoto(cameraController) { imageProxy ->
                        viewModel.processPhoto(imageProxy)
                        showTagDialog = true
                    }
                }
            )
        }
    }
    
    // Tag Dialog
    if (showTagDialog) {
        ComplianceTagDialog(
            onComplianceSelected = { status ->
                viewModel.setComplianceStatus(status)
            },
            onTagsSelected = { tags ->
                viewModel.applyTags(tags)
                viewModel.queueForAIAnalysis()
            }
        )
    }
}

// Project Selector Component
@Composable
fun ProjectQuickSelector(
    currentProject: Project?,
    onProjectChange: (Project) -> Unit
) {
    var expanded by remember { mutableStateOf(false) }
    val recentProjects = viewModel.recentProjects.collectAsState()
    
    Surface(
        modifier = Modifier.fillMaxWidth(),
        color = MaterialTheme.colors.primary
    ) {
        Row(
            modifier = Modifier
                .clickable { expanded = true }
                .padding(16.dp),
            verticalAlignment = Alignment.CenterVertically
        ) {
            Icon(Icons.Default.Construction, tint = Color.White)
            Spacer(modifier = Modifier.width(8.dp))
            Text(
                text = currentProject?.name ?: "Tap to select project",
                color = Color.White,
                style = MaterialTheme.typography.h6
            )
            Spacer(modifier = Modifier.weight(1f))
            Icon(Icons.Default.ArrowDropDown, tint = Color.White)
        }
        
        DropdownMenu(
            expanded = expanded,
            onDismissRequest = { expanded = false }
        ) {
            recentProjects.value.forEach { project ->
                DropdownMenuItem(
                    onClick = {
                        onProjectChange(project)
                        expanded = false
                    }
                ) {
                    Text(project.name)
                }
            }
            Divider()
            DropdownMenuItem(onClick = { showNewProjectDialog() }) {
                Icon(Icons.Default.Add)
                Text("New Project")
            }
        }
    }
}
