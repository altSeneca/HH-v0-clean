Pinch-to-Zoom Implementation

@Composable fun ZoomGestureHandler( cameraController: LifecycleCameraController, onZoomChange: (Float) -> Unit, modifier: Modifier = Modifier ) { Box( modifier = modifier.pointerInput(Unit) { detectTransformGestures { _, _, zoom, _ -> val currentZoom = cameraController.cameraInfo?.zoomState?.value?.zoomRatio ?: 1f val newZoom = (currentZoom * zoom).coerceIn(0.5f, 5.0f) cameraController.setZoomRatio(newZoom) onZoomChange(newZoom) } } ) }
